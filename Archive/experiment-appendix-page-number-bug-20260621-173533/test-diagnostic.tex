\documentclass[aspectratio=169]{beamer}

\makeatletter
\setbeamertemplate{footline}{%
  \hfill%
  \usebeamercolor[fg]{page number in head/foot}%
  \usebeamerfont{page number in head/foot}%
  % --- 诊断输出到 .log 文件 ---
  \typeout{=== FOOTLINE DEBUG ===}%
  \typeout{  framenumber       = \insertframenumber}%
  \typeout{  mainframenumber   = \insertmainframenumber}%
  \typeout{  totalframenumber  = \inserttotalframenumber}%
  \typeout{  inappendix?       = \ifbeamer@inappendix YES\else NO\fi}%
  \typeout{  framenumberinappendix = \insertframenumberinappendix}%
  % --- 实际显示 ---
  \ifbeamer@inappendix
    A-\insertframenumberinappendix
  \else
    \insertframenumber\,/\,\insertmainframenumber
  \fi
  \hspace{1.5em}\vspace{4pt}%
}
\makeatother
